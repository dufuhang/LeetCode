## 思路
今天的状态可由昨天的状态而来，可使用动态规划
本题和小偷家族的三道题类似:
- [198-打家劫舍](Dynamic-Programming/198-HouseRobber.md)
- [213-打家劫舍II]()
- [337-打家劫舍III]()
```cpp
//时间复杂度O(n)
//空间复杂度O(n)
class Solution 
{
public:
    int massage(vector<int>& nums) 
    {
        int count = nums.size();
        //没有预约的特殊情况
        if (count == 0) return 0;
        //dp数组
        //dp[i][0]代表今天不接预约的情况下的前i号最大预约数
        //dp[i][1]代表今天接预约的情况下的前i号最大预约数（包括今天）
        vector<vector<int> > dp(count, {0,0});
        for (int i = 0; i < count; i++)
        {
            //只有一个预约的时候，只能选择接受或者不接受
            if (i < 1)
            {
                dp[i][1] = nums[i];
                dp[i][0] = 0;
                continue;
            }
            //今天不接受预约：1、昨天接受了预约。2、昨天也没有接受预约
            dp[i][0] = max(dp[i-1][1], dp[i-1][0]);
            //今天接受了预约：1、昨天不能接受预约，因为中间要有休息的间隔
            dp[i][1] = dp[i-1][0] + nums[i];
        }
        //返回末尾的最大值
        return max(dp[count-1][1], dp[count-1][0]);
    }
};
```
> 今天的状态只依赖于昨天，所以只要有个变量存储昨天的状态即可，无需定义一个数组，空间复杂度降为O(1)
```cpp
//优化存储空间
//时间复杂度O(n)
//空间复杂度O(1)
class Solution 
{
public:
    int massage(vector<int>& nums) 
    {
        int count = nums.size();
        if (count == 0) return 0;
        int dp0 = 0;
        int dp1 = nums[0];
        int dp_prev_0 = 0;
        int dp_prev_1 = nums[0];
        for (int i = 1; i < count; i++)
        {
            dp0 = max(dp_prev_0, dp_prev_1);
            dp1 = dp_prev_0 + nums[i];
            dp_prev_0 = dp0;
            dp_prev_1 = dp1;
        }
        return max(dp0, dp1);
    }
};
```
